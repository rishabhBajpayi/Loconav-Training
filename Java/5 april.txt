class Team {

    String city;
    String department;
    Team(String c, String d){
        city = c;
        department = d;
    }

    @Override
    public final boolean equals(Object o) {
        // implementation
        if(o == this)
            return true;
        if( !(o instanceof Team) )
            return false;
        
        Team obj = (Team) o ;
        
        boolean cityequal = this.city == null && obj.city == null || this.city == obj.city;
        boolean departmentequal = this.department == null && obj.department == null || this.department == obj.department;
        
        return cityequal && departmentequal;
    }
}
